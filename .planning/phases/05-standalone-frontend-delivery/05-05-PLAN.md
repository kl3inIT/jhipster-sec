```typescript
  id?: number;
  authorityName: string;
  targetType: string;   // "ENTITY" | "ATTRIBUTE"
  target: string;       // "organization" or "organization.budget" or "organization.*"

}
```

Permission model mapping:
- Entity CRUD: targetType="ENTITY", target=entityCode, action=CREATE|READ|UPDATE|DELETE
- Attribute View: targetType="ATTRIBUTE", target="{entityCode}.{attrName}", action=VIEW
- Attribute Modify: targetType="ATTRIBUTE", target="{entityCode}.{attrName}", action=EDIT
- Wildcard attribute: targetType="ATTRIBUTE", target="{entityCode}.*", action=VIEW|EDIT

**Component class structure:**

```typescript
@Component({
  selector: 'app-permission-matrix',
  standalone: true,
  imports: [
    CommonModule, RouterModule,
    CardModule, TableModule, CheckboxModule, SplitterModule,
    ProgressSpinnerModule, ToastModule, ButtonModule,
  ],
  providers: [MessageService],
  templateUrl: './permission-matrix.component.html',
})
export default class PermissionMatrixComponent implements OnInit {
  // Injections
  private route = inject(ActivatedRoute);
  private catalogService = inject(SecCatalogService);
  private permissionService = inject(SecPermissionService);
  private messageService = inject(MessageService);

  // State
  roleName = '';
  catalogEntries: ISecCatalogEntry[] = [];
  loading = true;
  selectedEntity: ISecCatalogEntry | null = null;

  // Grant state: Map<"{target}:{action}", permissionId>
  // Key format: "organization:READ" for entity, "organization.budget:VIEW" for attribute
  private granted = new Map<string, number>();
```

**Initialization (ngOnInit):**

1. Extract `roleName` from route params: `this.route.snapshot.paramMap.get('name')`
2. Load catalog entries from `catalogService.query()` (store in `catalogEntries`)
3. Load permissions from `permissionService.query(roleName)` — build the `granted` Map:
   ```typescript
   permissions.forEach(p => {
     this.granted.set(`${p.target}:${p.action}`, p.id!);
   });
   ```
4. Set `loading = false` after both complete (use `forkJoin`)

**State query methods:**

```typescript
isGranted(target: string, action: string): boolean {
  return this.granted.has(`${target}:${action}`);
}

isWildcardGranted(entityCode: string, action: string): boolean {
  return this.granted.has(`${entityCode}.*:${action}`);
}
```

**Checkbox toggle handler:**

```typescript
togglePermission(targetType: string, target: string, action: string, checked: boolean): void {
  const key = `${target}:${action}`;
  if (checked) {
    // Optimistic: add to local state immediately
    const permission: ISecPermission = {
      authorityName: this.roleName,
      targetType,
      target,
      action,
      effect: 'GRANT',
    };
    this.permissionService.create(permission).subscribe({
      next: resp => {
        this.granted.set(key, resp.body!.id!);
      },
      error: () => {
        // Revert local state on error
        this.granted.delete(key);
        this.messageService.add({
          severity: 'error',
          summary: 'Save failed',
          detail: 'Could not update the permission. Please try again.',
        });
      },
    });
  } else {
    const id = this.granted.get(key);
    if (id == null) return;
    // Optimistic: remove from local state immediately
    this.granted.delete(key);
    this.permissionService.delete(id).subscribe({
      error: () => {
        // Revert: put it back on error
        this.granted.set(key, id);
        this.messageService.add({
          severity: 'error',
          summary: 'Save failed',
          detail: 'Could not update the permission. Please try again.',
        });
      },
    });
  }
}
```

Per D-23: each checkbox toggle fires individual POST/DELETE. No "Save All" button.
Per D-27: silent on success, toast on error.

**Attribute rows helper method:**

```typescript
getAttributeRows(entity: ISecCatalogEntry): { label: string; target: string; isWildcard: boolean }[] {
  const wildcard = { label: 'All attributes (*)', target: `${entity.code}.*`, isWildcard: true };
  const attrs = entity.attributes.map(attr => ({
    label: attr,
    target: `${entity.code}.${attr}`,
    isWildcard: false,
  }));
  return [wildcard, ...attrs];
}
```

The wildcard row is ALWAYS first (per UI-SPEC: "First row is always the wildcard row"). When wildcard is checked for VIEW or EDIT, individual rows for that action are `[disabled]="true"` (per D-25, UI-SPEC interaction contract, and Pitfall 4).
